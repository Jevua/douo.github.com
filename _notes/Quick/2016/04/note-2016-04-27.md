Gradle 自定义打包apk文件名格式

    android {
        ...
        buildTypes {
            release {
                ...
                applicationVariants.all { variant ->
                    variant.outputs.each { output ->
                        output.outputFile = new File(output.outputFile.parent,
                                defaultConfig.applicationId+"_"+defaultConfig.versionName+".apk");
                    }
                }
            }
        }
    }
